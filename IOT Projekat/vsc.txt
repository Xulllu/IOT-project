    function convertActiveLow(sensorReading) {
  const MAX_CURRENT = 0.020; // Maximum current in A (20 mA)
  const MAX_ADC_VALUE = 4095; // Maximum ADC value (12-bit)
  const R = 68; // Resistor value in ohms

  // Calculate the maximum voltage corresponding to the maximum ADC value
  const maxVoltage = 3.3; // Maximum voltage (V) for ESP32 ADC
  const maxCurrent = maxVoltage / R; // Maximum current (A) based on maximum voltage and resistor

  // Invert the sensor reading since it's in low active configuration
  const invertedReading = MAX_ADC_VALUE - sensorReading;

  // Scale the inverted sensor reading to the range of 0-20 mA
  const current = (invertedReading / MAX_ADC_VALUE) * maxCurrent;

  // Round the current to 4 decimal places
  const roundedCurrent = Math.round(current * 10000) / 10000; 

  return roundedCurrent.toFixed(4); // Return current with 4 decimal places
}



function convertVoltage(sensorReading) {
  const ADC_MAX = 4095; // Assuming 12-bit ADC
  const V_REF = 3.3; // Reference voltage of ESP32 ADC

  // Convert ADC reading to voltage (inverted)
  const voltage = ((ADC_MAX - sensorReading) * V_REF) / ADC_MAX;

  return voltage.toFixed(2); // Return voltage with 2 decimal places
}
