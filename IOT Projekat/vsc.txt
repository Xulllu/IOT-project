/* ================= LIVE DATA ================= */

let counter = 1;

firebase.database().ref('sensors/sensor1').on('value', (snapshot) => {
  const data = snapshot.val();

  const voltage = convertVoltage(parseFloat(data.napon));
  const current = convertActiveLow(parseFloat(data.jacinaS));
  const power = (current * voltage).toFixed(4);

  document.getElementById('napon').value += `${counter}. ${voltage} V\n`;
  document.getElementById('jacinaS').value += `${counter}. ${current} A\n`;
  document.getElementById('snaga').value += `${counter}. ${power} W\n`;

  counter++;
});

/* ================= HELPERS ================= */

function convertActiveLow(sensorReading) {
  const MAX_ADC = 4095;
  const V_REF = 3.3;
  const R = 68;

  const maxCurrent = V_REF / R;
  const inverted = MAX_ADC - sensorReading;
  const current = (inverted / MAX_ADC) * maxCurrent;

  return current.toFixed(4);
}

function convertVoltage(sensorReading) {
  const ADC_MAX = 4095;
  const V_REF = 3.3;

  const voltage = ((ADC_MAX - sensorReading) * V_REF) / ADC_MAX;
  return voltage.toFixed(2);
}
